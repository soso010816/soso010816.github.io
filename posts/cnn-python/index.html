				nn.ReLU(), # æ¿€æ´»å‡½æ•°ä¸º ReLU å‡½æ•°ï¼Œä¹Ÿå¯ä»¥æ ¹æ®éœ€è¦å¯¹å…¶è¿›è¡Œé‡æ–°é€‰æ‹©

				)
	
		# ç¬¬äºŒå±‚åï¼Œè½¬åŒ–å¾—åˆ°çš„ç»´åº¦
			h,w = findConv2dOutShape(h, w, nn.Conv2d(f, 2*f, kernel_size=3, padding=0, stride=2))
		h,w = h/2, w/2
