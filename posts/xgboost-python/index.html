def R_2(y, y_pred):
    y_mean = mean(y)
    sst = sum([(x-y_mean)**2 for x in y])
    ssr = sum([(x-y_mean)**2 for x in y_pred])
    sse = sum([(x-y)**2 for x,y in zip(y_pred, y)])
    return 1-sse/sst
